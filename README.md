## Install

```bash
npm install --save initialpageloader
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'initialpageloader'
import 'initialpageloader/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
